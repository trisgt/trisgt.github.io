#include <time.h>
#include <stdlib.h>

int are_raccoons_the_best(int racc){
	int r = rand(); // r is a completely random number, using the "rand()" function.

	r *= racc; // Ignore this line of code, it does nothing relevant.

	// Now we will check if r is an even or an odd number (50/50 chance):

	// As you will see, in theory there is only a 50% chance that raccoons are the best...

}

int main(){

	// You can modify the number as much as you want; raccoons will be the best any number of times:
	int repetitions = 10;

	srand(time(NULL));
